## 1. 代码质量

### 1.1 代码重复（Code Duplication）

**问题描述:**
6个 State Hook（useArrayState、useStackState、useQueueState、useLinkedListState、useTreeState、useGraphState）存在高度重复的样板代码：
- `addLog` 函数定义完全一致
- `handleUndo` / `handleRedo` 逻辑几乎相同
- `reset` 函数模式一致
- `logs`、`isAnimating` 状态管理重复

**具体位置:**
- [useArrayState.js](file:///d:/VibeCoding/数据结构学习助手3/src/hooks/useArrayState.js#L12-L16) vs [useStackState.js](file:///d:/VibeCoding/数据结构学习助手3/src/hooks/useStackState.js#L13-L16) vs [useQueueState.js](file:///d:/VibeCoding/数据结构学习助手3/src/hooks/useQueueState.js#L13-L16)

**影响:**
- 维护成本高：修改日志格式需要改6个文件
- 一致性风险：容易遗漏某些 Hook 的更新
- 代码膨胀：约 30% 的 Hook 代码是重复的

**优化方案:**
```javascript
// 提取通用 Hook 工厂函数
export function useDataStructureState(initialData, options = {}) {
  const { state: data, push, undo, redo, reset: resetHistory, canUndo, canRedo } = useHistory(initialData)
  const [logs, setLogs] = useState([])
  const [isAnimating, setIsAnimating] = useState(false)

  const addLog = useCallback((type, message) => {
    const time = new Date().toLocaleTimeString('zh-CN', { hour12: false })
    setLogs(prev => [...prev, { time, type, message }])
  }, [])

  const reset = useCallback(() => {
    resetHistory(initialData)
    setLogs([])
    showToast({ type: 'info', message: '已重置' })
  }, [resetHistory])

  const handleUndo = useCallback(() => {
    if (isAnimating || !canUndo()) return
    undo()
    addLog('info', '撤销上一步操作')
    showToast({ type: 'info', message: '已撤销' })
  }, [undo, canUndo, isAnimating, addLog])

  const handleRedo = useCallback(() => {
    if (isAnimating || !canRedo()) return
    redo()
    addLog('info', '重做操作')
    showToast({ type: 'info', message: '已重做' })
  }, [redo, canRedo, isAnimating, addLog])

  return {
    data, logs, isAnimating, setIsAnimating,
    push, addLog, reset,
    undo: handleUndo, redo: handleRedo,
    canUndo, canRedo,
  }
}
```

```javascript
// 使用 D3 数据绑定模式实现增量更新
export function renderArray(svg, data, options = {}) {
  const container = d3.select(svg)
  const { width, height } = options
  const { startX, startY } = layout(data, width, height)

  // 数据绑定 + join 模式
  const groups = container.selectAll('g.array-item')
    .data(data, (d, i) => i)
    .join(
      enter => enter.append('g')
        .attr('class', 'array-item')
        .attr('opacity', 0)
        .attr('transform', (d, i) => `translate(${startX + i * (RECT_WIDTH + GAP)}, ${startY - 20})`)
        .call(g => {
          g.append('rect').attr('width', RECT_WIDTH).attr('height', RECT_HEIGHT).attr('rx', 6)
            .attr('fill', '#3b82f6').attr('stroke', '#2563eb').attr('stroke-width', 2)
          g.append('text').attr('x', RECT_WIDTH / 2).attr('y', RECT_HEIGHT / 2).attr('dy', '0.35em')
            .attr('text-anchor', 'middle').attr('fill', 'white').attr('font-size', '16px').attr('font-weight', 'bold')
            .text(d => d)
          g.append('text').attr('x', RECT_WIDTH / 2).attr('y', RECT_HEIGHT + 18)
            .attr('text-anchor', 'middle').attr('fill', '#6b7280').attr('font-size', '12px')
            .text((d, i) => `[${i}]`)
        }),
      update => update
        .transition().duration(200)
        .attr('transform', (d, i) => `translate(${startX + i * (RECT_WIDTH + GAP)}, ${startY})`),
      exit => exit
```

### 2.4 排序算法动画阻塞主线程

**问题描述:**
- [useSortState.js:44-63](file:///d:/VibeCoding/数据结构学习助手3/src/hooks/useSortState.js#L44-L63) — 冒泡排序的动画在 for 循环中同步执行
- 大量 `await` 操作导致 React 无法响应其他状态更新
- 页面在排序期间完全冻结（除动画外）

**优化方案:**
```javascript
// 使用 requestAnimationFrame + 时间分片
export async function bubbleSortWithYield(arr, onStep, anim) {
  const n = arr.length
  for (let i = 0; i < n - 1; i++) {
    for (let j = 0; j < n - i - 1; j++) {
      if (anim?.isAborted?.()) return

      await onStep('compare', j, j + 1)

      if (arr[j] > arr[j + 1]) {
        await onStep('swap', j, j + 1)
        ;[arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
      }

      // 每 10 步让出主线程
      if (j % 10 === 0) {
        await new Promise(resolve => setTimeout(resolve, 0))
      }
    }
  }
}
```

## 4. 安全性

### 4.1 XSS 风险

**问题描述:**
- [arrayVisualizer.js:37](file:///d:/VibeCoding/数据结构学习助手3/src/visualizers/arrayVisualizer.js#L37) — `.text(d => d)` 虽然使用了 D3 的 `text()` 方法（自动转义），但其他位置可能使用 `.html()`
- 用户输入直接显示在 Toast 和 Log 中
- 如果未来添加 `.html()` 操作，存在 XSS 注入风险

**优化方案:**
```javascript
// 1. 创建安全的文本设置函数
export function safeSetText(selection, value) {
  // 始终使用 text() 而非 html()
  selection.text(String(value))
}

// 2. 输入验证和过滤
export function sanitizeInput(value, maxLength = 100) {
  if (typeof value !== 'string' && typeof value !== 'number') return ''
  const str = String(value).trim()
  if (str.length > maxLength) return str.slice(0, maxLength)
  return str.replace(/[<>"']/g, '')
```

```typescript
interface VisualizerRenderProps {
```
