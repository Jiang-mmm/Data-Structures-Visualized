## Feature 2: Algorithm Performance Comparison Chart (Priority: HIGH)

### Overview
Add a D3-based bar chart to SortComparePage that visually compares the 6 sorting algorithms' performance metrics side by side after execution completes.

### Task 2.1: Create PerformanceChart.jsx component

**File:** `src/components/PerformanceChart.jsx`

```jsx
import { useRef, useEffect } from 'react'
import { memo } from 'react'
import { useTheme } from '../hooks/useTheme'
import {
  select,
  selectAll,
} from '../utils/d3Imports'

const METRIC_COLORS = {
  comparisons: '#3b82f6',  // blue
  swaps: '#f43f5e',        // rose
  steps: '#8b5cf6',        // violet
}

const METRIC_LABELS = {
  comparisons: '比较次数',
  swaps: '交换次数',
  steps: '总步数',
}

const METRIC_LABELS_EN = {
  comparisons: 'Comparisons',
  swaps: 'Swaps',
  steps: 'Steps',
}

function PerformanceChart({ results, lang }) {
  const svgRef = useRef(null)
  const containerRef = useRef(null)
  const { mode } = useTheme()

  useEffect(() => {
    if (!svgRef.current || !results || Object.keys(results).length === 0) return

    const container = containerRef.current
    if (!container) return

    const rect = container.getBoundingClientRect()
    const width = rect.width
    const height = rect.height

    const margin = { top: 30, right: 20, bottom: 50, left: 60 }
    const innerW = width - margin.left - margin.right
    const innerH = height - margin.top - margin.bottom

    const svg = select(svgRef.current)
    svg.selectAll('*').remove()
    svg.attr('viewBox', `0 0 ${width} ${height}`)

    const g = svg.append('g')
      .attr('transform', `translate(${margin.left},${margin.top})`)

    const algoKeys = Object.keys(results)
    if (algoKeys.length === 0) return

    const metrics = ['comparisons', 'swaps', 'steps']
    const groupWidth = innerW / algoKeys.length
    const barWidth = groupWidth / (metrics.length + 1)

    const maxVal = Math.max(
      ...algoKeys.map(k => Math.max(metrics.map(m => results[k][m] || 0)))
    )
    if (maxVal === 0) return

    const yScale = (val) => innerH - (val / maxVal) * innerH

    // Y axis
    const yTicks = 5
    for (let i = 0; i <= yTicks; i++) {
      const val = Math.round((maxVal / yTicks) * i)
      const y = yScale(val)

      g.append('line')
        .attr('x1', 0).attr('x2', innerW)
        .attr('y1', y).attr('y2', y)
        .attr('stroke', mode === 'dark' ? '#334155' : '#e2e8f0')
        .attr('stroke-width', 1)

      g.append('text')
        .attr('x', -8).attr('y', y + 4)
        .attr('text-anchor', 'end')
        .attr('font-size', '9px')
        .attr('font-family', 'monospace')
        .attr('fill', mode === 'dark' ? '#94a3b8' : '#64748b')
        .text(val)
    }

    // Bars
    algoKeys.forEach((key, i) => {
      const groupX = i * groupWidth
      const algoResult = results[key]

      metrics.forEach((metric, j) => {
        const val = algoResult[metric] || 0
        const barH = (val / maxVal) * innerH
        const x = groupX + (j + 0.5) * barWidth
        const y = innerH - barH

        g.append('rect')
          .attr('x', x)
          .attr('y', y)
          .attr('width', barWidth - 2)
          .attr('height', Math.max(0, barH))
          .attr('fill', METRIC_COLORS[metric])
          .attr('rx', 2)
          .attr('stroke', mode === 'dark' ? '#1e293b' : '#fff')
          .attr('stroke-width', 1)

        // Value label on top
        if (barH > 15) {
          g.append('text')
            .attr('x', x + (barWidth - 2) / 2)
            .attr('y', y - 4)
            .attr('text-anchor', 'middle')
            .attr('font-size', '8px')
            .attr('font-family', 'monospace')
            .attr('fill', mode === 'dark' ? '#e2e8f0' : '#1e293b')
            .text(val)
        }
      })

      // Algorithm name label
      g.append('text')
        .attr('x', groupX + groupWidth / 2)
        .attr('y', innerH + 14)
        .attr('text-anchor', 'middle')
        .attr('font-size', '9px')
        .attr('font-weight', 'bold')
        .attr('fill', mode === 'dark' ? '#e2e8f0' : '#1e293b')
        .text(key.substring(0, 4))
    })

    // Legend
    const legendY = -10
    let legendX = 0
    metrics.forEach((metric) => {
      g.append('rect')
        .attr('x', legendX)
        .attr('y', legendY - 6)
        .attr('width', 8).attr('height', 8)
        .attr('fill', METRIC_COLORS[metric])
        .attr('rx', 1)

      g.append('text')
        .attr('x', legendX + 12)
        .attr('y', legendY)
        .attr('font-size', '9px')
        .attr('font-family', 'monospace')
        .attr('fill', mode === 'dark' ? '#94a3b8' : '#64748b')
        .text(lang === 'en' ? METRIC_LABELS_EN[metric] : METRIC_LABELS[metric])

      legendX += (lang === 'en' ? METRIC_LABELS_EN[metric].length : METRIC_LABELS[metric].length) * 6 + 30
    })

  }, [results, mode, lang])

  return (
    <div ref={containerRef} className="w-full h-48 neo-border bg-white dark:bg-slate overflow-hidden">
      <svg ref={svgRef} className="w-full h-full" />
    </div>
  )
}

export default memo(PerformanceChart)
```

```jsx
import { memo, useRef, useCallback } from 'react'
import { useGlobalSettings } from '../hooks/useGlobalSettings'

const ICONS = {
  insert: '+',
  delete: '−',
  push: '+',
  pop: '−',
  enqueue: '+',
  dequeue: '−',
  search: '🔍',
  reset: '↺',
  randomize: '🎲',
  swap: '⇄',
  undo: '↩',
  redo: '↪',
  default: '•',

function getOperationIcon(type) {
  if (!type) return ICONS.default
  const lower = type.toLowerCase()
  for (const [key, icon] of Object.entries(ICONS)) {
    if (lower.includes(key)) return icon
  }
  return ICONS.default
}

function TimelineItem({ entry, index, isCurrent, onClick }) {

  return (

      </span>

      <span className="text-[10px] opacity-60 flex-shrink-0">{index}</span>
    </button>
  )
}
```

### Task 3.4: Integrate Timeline into data structure pages

The timeline replaces or supplements the LogPanel. For this iteration, we add it as an **additional component** alongside LogPanel.

**Pattern for each page:**

```jsx
import Timeline from '../components/Timeline'

// The page needs to build a history entries array from logs or operations.
// Each log entry should have: { type, description }
// The currentIndex should track where we are in history.

// In ArrayPage.jsx, for example:
const [historyEntries, setHistoryEntries] = useState([])
const [historyIndex, setHistoryIndex] = useState(-1)

// Wrap each operation to add a history entry:
const handleInsert = async () => {
  // ... existing code ...
  insert(value, index)
  setHistoryEntries(prev => [...prev.slice(0, historyIndex + 1), {
    type: 'insert',
    description: `插入 ${value} 到索引 ${index}`,
  }])
  setHistoryIndex(prev => prev + 1)
  // ...
}

// Add jump handler:
const handleHistoryJump = (index) => {
  // Use the existing undo/redo mechanism
  // Jump forward or backward as needed
  while (historyIndex > index) {
    undo()
    setHistoryIndex(prev => prev - 1)
  }
  while (historyIndex < index) {
    redo()
    setHistoryIndex(prev => prev + 1)
  }
}

// In JSX, add after LogPanel:
<Timeline
  history={historyEntries}
  currentIndex={historyIndex}
  onJump={handleHistoryJump}
/>
```

**Alternative simpler approach:** Since each page already has `undo`/`redo` via `useDataStructureState`, we can derive history from the `logs` array. The logs already contain `{ time, type, message }`.

```jsx
// Derive history entries from logs:
const historyEntries = logs.map(log => ({
  type: log.type,
  description: log.message,
}))

// For currentIndex, we need to track it separately since logs only grow.
// Use a ref to track the "active" log position.
const [activeHistoryIndex, setActiveHistoryIndex] = useState(-1)
```

Given the complexity of integrating with existing state management, the recommended approach is:

**Use logs as the data source** — each log entry is a timeline node. Clicking a node triggers the appropriate number of undo/redo calls.

**Implementation per page (using ArrayPage as example):**

```jsx
// Add state for tracking undo count
const undoCountRef = useRef(0)

// After each operation, increment:
const handleInsert = async () => {
  // ...
  insert(value, index)
  undoCountRef.current = 0  // reset undo count on new operation
  // ...
}

// After undo:
const handleUndo = () => {
  if (!canUndo() || isAnimating) return
  undo()
  undoCountRef.current++
}

// After redo:
const handleRedo = () => {
  if (!canRedo() || isAnimating) return
  redo()
  undoCountRef.current--
}

// Timeline props:
<Timeline
  history={logs.map(log => ({ type: log.type, description: log.message }))}
  currentIndex={Math.max(0, logs.length - 1 - undoCountRef.current)}
  onJump={(targetIndex) => {
    const currentIndex = Math.max(0, logs.length - 1 - undoCountRef.current)
    if (targetIndex < currentIndex) {
      // Need to undo
      const count = currentIndex - targetIndex
      for (let i = 0; i < count; i++) {
        if (canUndo()) {
          undo()
          undoCountRef.current++
        }
      }
    } else if (targetIndex > currentIndex) {
      // Need to redo
      const count = targetIndex - currentIndex
      for (let i = 0; i < count; i++) {
        if (canRedo()) {
          redo()
          undoCountRef.current--
        }
      }
    }
  }}
/>
```

Given the complexity and risk of this approach (multiple undo/redo calls can be fragile), **a simpler alternative** for v3.9:

**The Timeline only shows the history visually and triggers single undo/redo per click.** Clicking a past node triggers a single undo; clicking a future node triggers a single redo. Full jump-to-state is a future enhancement.

```jsx
// Simpler approach: Timeline shows history, clicking any past entry = undo,
// clicking the next entry = redo
<Timeline
  history={logs.map(log => ({ type: log.type, description: log.message }))}
  currentIndex={logs.length - 1 - undoCountRef.current}
  onJump={(targetIndex) => {
    const current = Math.max(0, logs.length - 1 - undoCountRef.current)
    if (targetIndex < current && canUndo()) {
      undo()
      undoCountRef.current++
    } else if (targetIndex > current && canRedo()) {
      redo()
      undoCountRef.current--
    }
  }}
/>
```

This is the recommended approach for v3.9. Full jump-to-state requires snapshot-based history which is a larger refactor.
