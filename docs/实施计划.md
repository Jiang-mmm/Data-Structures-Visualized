```javascript
import { useRef, useEffect, useState } from 'react'

export function useVisualizer() {
  const containerRef = useRef(null)
  const svgRef = useRef(null)
  const [dimensions, setDimensions] = useState({ width: 800, height: 400 })

  useEffect(() => {
    const updateDimensions = () => {
      if (containerRef.current) {
        const { width, height } = containerRef.current.getBoundingClientRect()
        setDimensions({ width, height })
      }
    }

    updateDimensions()
    window.addEventListener('resize', updateDimensions)
    return () => window.removeEventListener('resize', updateDimensions)
  }, [])

  useEffect(() => {
    if (containerRef.current && !svgRef.current) {
      svgRef.current = containerRef.current
        .appendChild(document.createElementNS('http://www.w3.org/2000/svg', 'svg'))
      svgRef.current.setAttribute('width', dimensions.width)
      svgRef.current.setAttribute('height', dimensions.height)
    }

    if (svgRef.current) {
      svgRef.current.setAttribute('width', dimensions.width)
      svgRef.current.setAttribute('height', dimensions.height)
    }
  }, [dimensions])

  return { containerRef, svgRef, dimensions }
}
```

```javascript
import * as d3 from 'd3'

const RECT_WIDTH = 60
const RECT_HEIGHT = 50
const GAP = 10
const ANIMATION_DURATION = 500

export function renderArray(svg, data, options = {}) {
  const { width, height } = options
  const container = d3.select(svg)

  container.selectAll('*').remove()

  const startX = (width - (data.length * (RECT_WIDTH + GAP) - GAP)) / 2
  const startY = (height - RECT_HEIGHT) / 2

  const groups = container.selectAll('g.array-item')
    .data(data, (d, i) => i)
    .join('g')
    .attr('class', 'array-item')
    .attr('transform', (d, i) => `translate(${startX + i * (RECT_WIDTH + GAP)}, ${startY})`)

  groups.append('rect')
    .attr('width', RECT_WIDTH)
    .attr('height', RECT_HEIGHT)
    .attr('rx', 6)
    .attr('fill', '#3b82f6')
    .attr('stroke', '#2563eb')
    .attr('stroke-width', 2)

  groups.append('text')
    .attr('x', RECT_WIDTH / 2)
    .attr('y', RECT_HEIGHT / 2)
    .attr('dy', '0.35em')
    .attr('text-anchor', 'middle')
    .attr('fill', 'white')
    .attr('font-size', '16px')
    .attr('font-weight', 'bold')
    .text(d => d)

  groups.append('text')
    .attr('x', RECT_WIDTH / 2)
    .attr('y', RECT_HEIGHT + 18)
    .attr('text-anchor', 'middle')
    .attr('fill', '#6b7280')
    .attr('font-size', '12px')
    .text((d, i) => i)
}

export function animateInsert(svg, index, value, options = {}) {
  return new Promise((resolve) => {
    const container = d3.select(svg)
    // 实现插入动画逻辑
    setTimeout(resolve, ANIMATION_DURATION)

}

  return new Promise((resolve) => {

}
```

```jsx
import { useState, useCallback } from 'react'
import OperationBar from '../components/OperationBar'
import Visualizer from '../components/Visualizer'
import LogPanel from '../components/LogPanel'
import { renderArray, animateInsert, animateDelete, animateSearch } from '../visualizers/arrayVisualizer'

export default function ArrayPage() {
  const [data, setData] = useState([8, 3, 12, 5, 9])
  const [inputValue, setInputValue] = useState('')
  const [inputIndex, setInputIndex] = useState('')
  const [logs, setLogs] = useState([])
  const [isAnimating, setIsAnimating] = useState(false)

  const addLog = useCallback((type, message) => {
    const time = new Date().toLocaleTimeString('zh-CN', { hour12: false })
    setLogs(prev => [...prev, { time, type, message }])
  }, [])

  const handleInsert = async () => {
    if (isAnimating || !inputValue || !inputIndex) return
    const value = parseInt(inputValue)
    const index = parseInt(inputIndex)
    if (isNaN(value) || isNaN(index)) return

    setIsAnimating(true)
    addLog('oper', `insert(${value}, ${index}) - 开始`)

    const newData = [...data]
    newData.splice(index, 0, value)
    setData(newData)
    addLog('oper', `insert(${value}, ${index}) - 完成，数组长度: ${newData.length}`)
    addLog('code', `array.splice(${index}, 0, ${value});`)

    setIsAnimating(false)
    setInputValue('')
    setInputIndex('')
  }

  const handleDelete = async () => {
    if (isAnimating || !inputIndex) return
    const index = parseInt(inputIndex)
    if (isNaN(index) || index < 0 || index >= data.length) return

    setIsAnimating(true)
    const value = data[index]
    addLog('oper', `delete(${index}) - 删除值: ${value}`)

    const newData = [...data]
    newData.splice(index, 1)
    setData(newData)
    addLog('oper', `delete(${index}) - 完成，数组长度: ${newData.length}`)
    addLog('code', `array.splice(${index}, 1);`)

    setIsAnimating(false)
    setInputIndex('')
  }

  const handleSearch = async () => {
    if (isAnimating || !inputValue) return
    const value = parseInt(inputValue)
    if (isNaN(value)) return

    setIsAnimating(true)
    addLog('oper', `search(${value}) - 开始查找`)

    const index = data.indexOf(value)
    if (index !== -1) {
      addLog('oper', `search(${value}) - 找到，索引: ${index}`)
    } else {
      addLog('oper', `search(${value}) - 未找到`)
    }

    setIsAnimating(false)
    setInputValue('')
  }

  const handleRandom = () => {
    const newData = Array.from({ length: 8 }, () => Math.floor(Math.random() * 99) + 1)
    setData(newData)
    addLog('info', `随机生成数组: [${newData.join(', ')}]`)
  }

  const handleReset = () => {
    setData([8, 3, 12, 5, 9])
    setLogs([])
  }

  return (
    <div className="flex flex-col h-screen">
      <header className="bg-white dark:bg-gray-800 border-b border-gray-200 dark:border-gray-700 px-6 py-4">
        <div className="flex items-center justify-between">
          <div>
            <h1 className="text-2xl font-bold text-gray-800 dark:text-gray-100">数组 Array</h1>
            <p className="text-sm text-gray-500 dark:text-gray-400 mt-1">连续内存存储，支持随机访问</p>
          </div>
          <div className="flex items-center gap-3">
            <button onClick={handleReset} className="px-3 py-1.5 text-sm border border-gray-300 dark:border-gray-600 rounded-lg hover:bg-gray-50 dark:hover:bg-gray-700 transition-colors text-gray-700 dark:text-gray-300">重置</button>
            <button onClick={handleRandom} className="px-3 py-1.5 text-sm bg-primary-600 text-white rounded-lg hover:bg-primary-700 transition-colors">随机数据</button>
          </div>
        </div>
      </header>

      <OperationBar>
        <span className="text-sm font-medium text-gray-600 dark:text-gray-300">操作:</span>
        <input
          type="number"
          placeholder="值"
          value={inputValue}
          onChange={(e) => setInputValue(e.target.value)}
          className="w-20 px-3 py-1.5 text-sm border border-gray-300 dark:border-gray-600 rounded-lg bg-white dark:bg-gray-700 focus:ring-2 focus:ring-primary-500 outline-none text-gray-800 dark:text-gray-100"
        />
        <input
          type="number"
          placeholder="索引"
          value={inputIndex}
          onChange={(e) => setInputIndex(e.target.value)}
```

```jsx
import { useState, useCallback, useRef } from 'react'
import OperationBar from '../components/OperationBar'
import Visualizer from '../components/Visualizer'
import LogPanel from '../components/LogPanel'
import { renderSortBars } from '../visualizers/sortVisualizer'

export default function SortPage() {
  const [data, setData] = useState([64, 34, 25, 12, 22, 11, 90, 5])
  const [logs, setLogs] = useState([])
  const [isSorting, setIsSorting] = useState(false)
  const [speed, setSpeed] = useState(500)
  const stopRef = useRef(false)

  const addLog = useCallback((type, message) => {
    const time = new Date().toLocaleTimeString('zh-CN', { hour12: false })
    setLogs(prev => [...prev, { time, type, message }])
  }, [])

  const sleep = (ms) => new Promise(resolve => setTimeout(resolve, ms))

  const bubbleSort = async () => {
    setIsSorting(true)
    stopRef.current = false
    addLog('info', '开始冒泡排序')
    const arr = [...data]
    const n = arr.length

    for (let i = 0; i < n - 1; i++) {
      for (let j = 0; j < n - i - 1; j++) {
        if (stopRef.current) {
          addLog('info', '排序已停止')
          setIsSorting(false)
          return
        }

        addLog('oper', `比较: arr[${j}]=${arr[j]} 和 arr[${j + 1}]=${arr[j + 1}]`)
        await sleep(speed)

        if (arr[j] > arr[j + 1]) {
          [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]]
          setData([...arr])
          addLog('oper', `交换: arr[${j}] ↔ arr[${j + 1}]`)
          await sleep(speed)
        }
      }
    }

    addLog('info', '冒泡排序完成')
    setIsSorting(false)
  }

  const selectionSort = async () => {
    setIsSorting(true)
    stopRef.current = false
    addLog('info', '开始选择排序')
    const arr = [...data]
    const n = arr.length

    for (let i = 0; i < n - 1; i++) {
      let minIdx = i
      for (let j = i + 1; j < n; j++) {
        if (stopRef.current) {
          addLog('info', '排序已停止')
          setIsSorting(false)
          return
        }

        if (arr[j] < arr[minIdx]) {
          minIdx = j
        }
      }
      if (minIdx !== i) {
        [arr[i], arr[minIdx]] = [arr[minIdx], arr[i]]
        setData([...arr])
        addLog('oper', `交换: arr[${i}]=${arr[i]} ↔ arr[${minIdx}]=${arr[minIdx]}`)
        await sleep(speed)
      }
    }

    addLog('info', '选择排序完成')
    setIsSorting(false)
  }

  const insertionSort = async () => {
    setIsSorting(true)
    stopRef.current = false
    addLog('info', '开始插入排序')
    const arr = [...data]
    const n = arr.length

    for (let i = 1; i < n; i++) {
      const key = arr[i]
      let j = i - 1

      while (j >= 0 && arr[j] > key) {
        if (stopRef.current) {
          addLog('info', '排序已停止')
          setIsSorting(false)
          return
        }

        arr[j + 1] = arr[j]
        setData([...arr])
        addLog('oper', `移动: arr[${j}] → arr[${j + 1}]`)
        await sleep(speed)
        j--
      }
      arr[j + 1] = key
      setData([...arr])
      addLog('oper', `插入: ${key} 到位置 ${j + 1}`)
      await sleep(speed)
    }

    setIsSorting(false)
  }

  const handleStop = () => {
    stopRef.current = true
  }

  const handleRandom = () => {
    const newData = Array.from({ length: 10 }, () => Math.floor(Math.random() * 99) + 1)
    setData(newData)
    setLogs([])
  }

  return (
    <div className="flex flex-col h-screen">
```

```javascript
import * as d3 from 'd3'

const NODE_RADIUS = 25
const GAP = 80

export function renderLinkedList(svg, data, options = {}) {
  const { width, height } = options
  const container = d3.select(svg)

  container.selectAll('*').remove()

  const totalWidth = data.length * (NODE_RADIUS * 2 + GAP) - GAP
  const startX = (width - totalWidth) / 2 + NODE_RADIUS
  const startY = height / 2

  // 绘制连线
  for (let i = 0; i < data.length - 1; i++) {
    const x1 = startX + i * (NODE_RADIUS * 2 + GAP) + NODE_RADIUS
    const x2 = startX + (i + 1) * (NODE_RADIUS * 2 + GAP) - NODE_RADIUS

    container.append('line')
      .attr('x1', x1)
      .attr('y1', startY)
      .attr('x2', x2)
      .attr('y2', startY)
      .attr('stroke', '#6b7280')
      .attr('stroke-width', 2)

    // 箭头
    container.append('polygon')
      .attr('points', `${x2},${startY} ${x2 - 8},${startY - 5} ${x2 - 8},${startY + 5}`)
      .attr('fill', '#6b7280')
  }

  // 绘制节点
  const groups = container.selectAll('g.list-node')
    .data(data)
    .join('g')
    .attr('transform', (d, i) => `translate(${startX + i * (NODE_RADIUS * 2 + GAP)}, ${startY})`)

  groups.append('circle')
    .attr('r', NODE_RADIUS)
    .attr('fill', '#3b82f6')
    .attr('stroke', '#2563eb')
    .attr('stroke-width', 2)

    .attr('text-anchor', 'middle')
    .attr('fill', 'white')
    .attr('font-size', '14px')
    .attr('font-weight', 'bold')
    .text(d => d)

  // 头节点标记
  if (data.length > 0) {
    container.append('text')
      .attr('x', startX)
      .attr('y', startY - NODE_RADIUS - 15)
      .attr('text-anchor', 'middle')
```

```javascript
import * as d3 from 'd3'

const NODE_RADIUS = 22

function buildTreeData(values) {
  if (values.length === 0) return null

  const root = { value: values[0], children: [] }
  const queue = [root]
  let i = 1

  while (queue.length > 0 && i < values.length) {
    const node = queue.shift()

    if (i < values.length) {
      const left = { value: values[i], children: [] }
      node.children.push(left)
      queue.push(left)
      i++
    }

    if (i < values.length) {
      const right = { value: values[i], children: [] }
      node.children.push(right)
      queue.push(right)
      i++
    }
  }

  return root
}

export function renderTree(svg, data, options = {}) {
  const { width, height } = options
  const container = d3.select(svg)

  container.selectAll('*').remove()

  if (data.length === 0) return

  const root = d3.hierarchy(buildTreeData(data))
  const treeLayout = d3.tree().size([width - 80, height - 120])
  treeLayout(root)

  const g = container.append('g')
    .attr('transform', 'translate(40, 40)')

  // 绘制连线
  g.selectAll('path.link')
    .data(root.links())
    .join('path')
    .attr('class', 'link')
    .attr('fill', 'none')
    .attr('stroke', '#6b7280')
    .attr('stroke-width', 2)
    .attr('d', d3.linkVertical()
      .x(d => d.x)
      .y(d => d.y))

  // 绘制节点
  const nodes = g.selectAll('g.tree-node')
    .data(root.descendants())
    .join('g')
    .attr('class', 'tree-node')
    .attr('transform', d => `translate(${d.x}, ${d.y})`)

  nodes.append('circle')
    .attr('r', NODE_RADIUS)
    .attr('fill', '#10b981')
    .attr('stroke', '#059669')
    .attr('stroke-width', 2)

  nodes.append('text')
    .attr('dy', '0.35em')
    .attr('text-anchor', 'middle')
    .attr('fill', 'white')
    .attr('font-size', '13px')
    .attr('font-weight', 'bold')
    .text(d => d.data.value)
}
```

### Task 11: GitHub Pages 部署配置

**Files:**
- Create: `.github/workflows/deploy.yml`

- [ ] **Step 1: 创建 GitHub Actions 部署配置**

Create: `.github/workflows/deploy.yml`
```yaml
name: Deploy to GitHub Pages

on:
  push:
    branches: [main]
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: 'pages'
  cancel-in-progress: true

jobs:
  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build

      - name: Setup Pages
        uses: actions/configure-pages@v4

      - name: Upload artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: './dist'

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
```
