**Step 1: 创建 useVisualizer Hook（优化版）**

Create: `src/hooks/useVisualizer.js`
```javascript
import { useRef, useEffect, useState, useCallback } from 'react'

export function useVisualizer() {
  const containerRef = useRef(null)
  const svgRef = useRef(null)
  const [dimensions, setDimensions] = useState({ width: 800, height: 400 })

  const updateDimensions = useCallback(() => {
    if (containerRef.current) {
      const { width, height } = containerRef.current.getBoundingClientRect()
      setDimensions({
        width: Math.floor(width),
        height: Math.floor(height)
      })
    }
  }, [])

  useEffect(() => {
    updateDimensions()

    const observer = new ResizeObserver(() => {
      updateDimensions()
    })

    if (containerRef.current) {
      observer.observe(containerRef.current)
    }

    return () => {
      observer.disconnect()
    }
  }, [updateDimensions])

  return { containerRef, svgRef, dimensions }
}
```

**Step 2: 创建 Visualizer 组件（优化版）**

```jsx
import { useEffect } from 'react'
import { useVisualizer } from '../hooks/useVisualizer'

export default function Visualizer({ data, renderFn, className = '' }) {
  const { containerRef, svgRef, dimensions } = useVisualizer()

  useEffect(() => {
    if (svgRef.current && renderFn && data) {
      try {
        renderFn(svgRef.current, data, dimensions)
      } catch (error) {
        console.error('Visualization render error:', error)
      }
    }
  }, [data, renderFn, svgRef, dimensions])

  return (
    <div
      ref={containerRef}
      className={`
        flex-1 bg-gradient-to-br from-blue-50/50 to-purple-50/50
        dark:from-blue-900/10 dark:to-purple-900/10
        relative min-h-[400px] overflow-hidden
        ${className}
      `}
    >
      <svg
        ref={svgRef}
        width={dimensions.width}
        height={dimensions.height}
        className="w-full h-full"
      />
      {(!data || data.length === 0) && (
        <div className="absolute inset-0 flex items-center justify-center">
          <p className="text-gray-400 dark:text-gray-500 text-sm">
            暂无数据，请添加元素
          </p>
        </div>
      )}
    </div>
  )
}
```

```jsx
const typeConfig = {
  info: { color: 'text-green-400', bg: 'bg-green-400/10', label: 'INFO' },
  oper: { color: 'text-blue-400', bg: 'bg-blue-400/10', label: 'OPER' },

      scrollRef.current.scrollTop = scrollRef.current.scrollHeight
    }
  }, [filteredLogs, autoScroll])

  const handleScroll = () => {
    if (scrollRef.current) {
      const { scrollTop, scrollHeight, clientHeight } = scrollRef.current
      setAutoScroll(scrollHeight - scrollTop - clientHeight < 50)
    }
  }

  const clearLogs = () => {
    // This would need to be passed from parent
  }

  return (
    <div className={`${maxHeight} bg-gray-900 text-gray-100 border-t border-gray-700 flex flex-col`}>
      <div className="flex items-center justify-between px-4 py-2 bg-gray-800 border-b border-gray-700">
        <div className="flex items-center gap-4">
          <span className="text-sm font-medium">执行日志</span>
          <div className="flex gap-1">
            {['all', 'oper', 'code', 'info', 'error'].map((f) => (
              <button

                onClick={() => setFilter(f)}
                className={`

                  ${filter === f

                    : 'text-gray-400 hover:bg-gray-700 hover:text-gray-200'
                  }
                `}
              >
                {f === 'all' ? '全部' : typeConfig[f]?.label || f}
              </button>
            ))}
          </div>
        </div>
        <div className="flex items-center gap-3">
          <span className="text-xs text-gray-500">
            {filteredLogs.length} 条记录
          </span>
          <button
            onClick={() => setAutoScroll(!autoScroll)}
            className={`text-xs px-2 py-0.5 rounded ${
              autoScroll ? 'bg-primary-600 text-white' : 'text-gray-400 hover:bg-gray-700'
            }`}
          >
            {autoScroll ? '自动滚动' : '手动滚动'}
          </button>
        </div>
      </div>

      <div
        ref={scrollRef}
        onScroll={handleScroll}

      >
        {filteredLogs.length === 0 ? (
          <div className="text-gray-500 text-center py-4">
            暂无日志记录
          </div>
        ) : (
          filteredLogs.map((log, i) => {

              <div
                key={i}
                className={`flex gap-2 py-0.5 px-2 rounded ${config.bg} animate-fade-in`}
```

```jsx
import { useState, useEffect, useCallback } from 'react'

let toastId = 0

const toastStore = {
  listeners: new Set(),
  toasts: [],
  add(toast) {
    const id = ++toastId
    this.toasts.push({ ...toast, id })
    this.notify()
    setTimeout(() => this.remove(id), toast.duration || 3000)
    return id
  },
  remove(id) {
    this.toasts = this.toasts.filter(t => t.id !== id)
    this.notify()
  },
  notify() {
    this.listeners.forEach(listener => listener([...this.toasts]))
  },
  subscribe(listener) {
    this.listeners.add(listener)
    return () => this.listeners.delete(listener)
  }
}

export function useToast() {
  const [toasts, setToasts] = useState([])

  useEffect(() => {
    return toastStore.subscribe(setToasts)
  }, [])

  const toast = useCallback((options) => {
    return toastStore.add(options)
  }, [])

  return { toasts, toast }
}

export function showToast(options) {
  return toastStore.add(options)
}

export default function ToastContainer() {
  const { toasts } = useToast()

  if (toasts.length === 0) return null

  const typeStyles = {
    success: 'bg-green-500',
    error: 'bg-red-500',
    warning: 'bg-amber-500',
    info: 'bg-blue-500',
  }

  return (
    <div className="fixed top-4 right-4 z-50 flex flex-col gap-2">
      {toasts.map((t) => (
        <div
          key={t.id}
          className={`
            px-4 py-3 rounded-lg text-white text-sm shadow-lg
            animate-slide-up
            ${typeStyles[t.type] || typeStyles.info}
          `}
        >
          {t.message}
```

```javascript
export function animateDelete(svg, index, data, options = {}) {
  return new Promise((resolve) => {

    const { width, height } = options

    const startY = (height - RECT_HEIGHT) / 2

      .transition()
      .duration(200)
      .attr('fill', '#ef4444')

      targetGroup
        .transition()
        .duration(ANIMATION_DURATION / 2)
        .attr('transform', `translate(${
          parseFloat(targetGroup.attr('transform').split('(')[1])

            .filter((d, i) => i !== index)

            .then(resolve)

export function animateSearch(svg, index, data, options = {}) {
  return new Promise((resolve) => {
    const container = d3.select(svg)

    const groups = container.selectAll('g.array-item')

    let delay = 0
    groups.each(function(d, i) {
      const group = d3.select(this)
      const isTarget = i === index
      const isFound = index !== -1

            .attr('fill', '#10b981')

            .transition()
            .duration(150)
            .attr('fill', '#3b82f6')
            .attr('stroke', '#2563eb')
        }
      }, delay)

      delay += isTarget ? 0 : 200
    })

  })
```

```javascript
import { useState, useCallback } from 'react'
import { showToast } from '../components/Toast'

const INITIAL_DATA = [8, 3, 12, 5, 9]

export function useArrayState() {
  const [data, setData] = useState(INITIAL_DATA)
  const [logs, setLogs] = useState([])
  const [isAnimating, setIsAnimating] = useState(false)

  const addLog = useCallback((type, message) => {
    const time = new Date().toLocaleTimeString('zh-CN', { hour12: false })
    setLogs(prev => [...prev, { time, type, message }])
  }, [])

  const insert = useCallback((value, index) => {
    if (index < 0 || index > data.length) {
      showToast({ type: 'error', message: `索引越界: ${index}` })
      addLog('error', `insert(${value}, ${index}) - 索引越界`)
      return false
    }

    const newData = [...data]
    newData.splice(index, 0, value)
    setData(newData)
    addLog('oper', `insert(${value}, ${index}) - 插入成功，数组长度: ${newData.length}`)
    addLog('code', `array.splice(${index}, 0, ${value});`)
    showToast({ type: 'success', message: `插入 ${value} 到位置 ${index}` })
    return true
  }, [data, addLog])

  const remove = useCallback((index) => {
    if (index < 0 || index >= data.length) {
      showToast({ type: 'error', message: `索引越界: ${index}` })
      addLog('error', `delete(${index}) - 索引越界`)
      return null
    }

    const value = data[index]
    const newData = [...data]
    newData.splice(index, 1)
    setData(newData)
    addLog('oper', `delete(${index}) - 删除值: ${value}，数组长度: ${newData.length}`)
    addLog('code', `array.splice(${index}, 1);`)
    showToast({ type: 'success', message: `删除位置 ${index} 的元素 ${value}` })
    return value
  }, [data, addLog])

  const search = useCallback((value) => {
    const index = data.indexOf(value)
    if (index !== -1) {
      addLog('oper', `search(${value}) - 找到，索引: ${index}`)
      showToast({ type: 'success', message: `找到 ${value} 在位置 ${index}` })
    } else {
      addLog('oper', `search(${value}) - 未找到`)
      showToast({ type: 'warning', message: `未找到 ${value}` })
    }
    return index
  }, [data, addLog])

  const randomize = useCallback(() => {
    const newData = Array.from({ length: 8 }, () => Math.floor(Math.random() * 99) + 1)
    setData(newData)
    addLog('info', `随机生成数组: [${newData.join(', ')}]`)
    showToast({ type: 'info', message: '已生成随机数组' })
  }, [addLog])

  const reset = useCallback(() => {
    setData(INITIAL_DATA)
    setLogs([])
    showToast({ type: 'info', message: '已重置' })
  }, [])

  return {
    data,
    logs,
    isAnimating,
    setIsAnimating,
    insert,
    remove,
    search,
    randomize,
    reset,
  }
}
```

```javascript
import { useState, useCallback } from 'react'
import { showToast } from '../components/Toast'

const INITIAL_DATA = [8, 17, 42]
const MAX_SIZE = 10

export function useStackState() {
  const [data, setData] = useState(INITIAL_DATA)
  const [logs, setLogs] = useState([])
  const [isAnimating, setIsAnimating] = useState(false)

  const addLog = useCallback((type, message) => {
    const time = new Date().toLocaleTimeString('zh-CN', { hour12: false })
    setLogs(prev => [...prev, { time, type, message }])
  }, [])

  const push = useCallback((value) => {
    if (data.length >= MAX_SIZE) {
      showToast({ type: 'error', message: `栈已满 (最大 ${MAX_SIZE})` })
      addLog('error', `push(${value}) - 栈满，无法入栈`)
      return false
    }

    const newData = [...data, value]
    setData(newData)
    addLog('oper', `push(${value}) → 栈顶: ${value}, 大小: ${newData.length}`)
    addLog('code', `stack[++top] = ${value};`)
    showToast({ type: 'success', message: `入栈 ${value}` })
    return true
  }, [data, addLog])

  const pop = useCallback(() => {
    if (data.length === 0) {
      showToast({ type: 'error', message: '栈为空，无法出栈' })
      addLog('error', 'pop() → 栈为空')
      return null
    }

    const value = data[data.length - 1]
    const newData = data.slice(0, -1)
    setData(newData)
    addLog('oper', `pop() → 返回值: ${value}, 大小: ${newData.length}`)
    addLog('code', `return stack[top--];`)
    showToast({ type: 'success', message: `出栈 ${value}` })
    return value
  }, [data, addLog])

  const peek = useCallback(() => {
    if (data.length === 0) {
      showToast({ type: 'warning', message: '栈为空' })
      addLog('error', 'peek() → 栈为空')
      return null
    }

    const value = data[data.length - 1]
    addLog('oper', `peek() → 栈顶: ${value}`)
    showToast({ type: 'info', message: `栈顶: ${value}` })
    return value
  }, [data, addLog])

  const clear = useCallback(() => {
    setData([])
    addLog('oper', 'clear() → 栈已清空')
    showToast({ type: 'info', message: '栈已清空' })
  }, [addLog])

  const reset = useCallback(() => {
    setData(INITIAL_DATA)
    setLogs([])
    showToast({ type: 'info', message: '已重置' })
  }, [])

  return {
    data,
    logs,
    isAnimating,
    setIsAnimating,
    push,
    pop,
    peek,
    clear,
    reset,
    size: data.length,
  }
}
```

```jsx
import { useState } from 'react'
import PageHeader from '../components/PageHeader'
import OperationBar, {
  OperationInput,
  OperationButton,
  OperationLabel,
  OperationInfo,
} from '../components/OperationBar'
import Visualizer from '../components/Visualizer'
import LogPanel from '../components/LogPanel'
import { renderStack, animatePush, animatePop } from '../visualizers/stackVisualizer'
import { useStackState } from '../hooks/useStackState'
import { useVisualizer } from '../hooks/useVisualizer'

export default function StackPage() {
  const {
    data,
    logs,
    isAnimating,
    setIsAnimating,
    push,
    pop,
    peek,
    clear,
    reset,
    size,
  } = useStackState()

  const { svgRef, dimensions } = useVisualizer()
  const [inputValue, setInputValue] = useState('')

  const handlePush = async () => {
    if (isAnimating || !inputValue) return
    const value = parseInt(inputValue)
    if (isNaN(value)) return

    setIsAnimating(true)
    try {
      if (svgRef.current) {
        await animatePush(svgRef.current, value, data, dimensions)
      }
      push(value)
    } catch (error) {
      console.error('Push animation error:', error)
    }
    setIsAnimating(false)
    setInputValue('')
  }

  const handlePop = async () => {
    if (isAnimating || data.length === 0) return

    setIsAnimating(true)
    try {
      if (svgRef.current) {
        await animatePop(svgRef.current, data, dimensions)
      }
      pop()
    } catch (error) {
      console.error('Pop animation error:', error)
    }
    setIsAnimating(false)
  }

  return (
    <div className="flex flex-col h-screen">
      <PageHeader
        title="栈 Stack"
        subtitle="后进先出 (LIFO) 的线性数据结构"
        icon="☰"
      >
        <OperationButton variant="outline" onClick={reset}>
          重置
        </OperationButton>
      </PageHeader>

      <OperationBar>
        <OperationLabel>操作:</OperationLabel>
        <OperationInput
          type="number"
          placeholder="输入值"
          value={inputValue}
          onChange={setInputValue}
        />
        <OperationButton
          variant="purple"
          onClick={handlePush}
          disabled={isAnimating}
        >
          + 入栈 Push
        </OperationButton>
        <OperationButton
          variant="danger"
          onClick={handlePop}
          disabled={isAnimating || data.length === 0}
        >
          - 出栈 Pop
        </OperationButton>
        <OperationButton
          variant="outline"
          onClick={peek}
        >
          Peek 查看栈顶
        </OperationButton>
        <OperationButton
          variant="outline"
          onClick={clear}
          disabled={data.length === 0}
        >
          清空
        </OperationButton>
        <OperationInfo>
          <span className="text-xs text-gray-500 dark:text-gray-400">
            大小: {size}
          </span>
        </OperationInfo>
      </OperationBar>

      <Visualizer data={data} renderFn={renderStack} />
      <LogPanel logs={logs} />
    </div>
  )
}
```

```javascript
import * as d3 from 'd3'
```
